DROP TABLE IF EXISTS STAFF;

DROP TABLE IF EXISTS Driver;


CREATE TABLE STAFF
(
  ORDER_ID INT PRIMARY KEY AUTO_INCREMENT,
  DESCR VARCHAR(32),
  ORDER_NAME  VARCHAR(64)

);


CREATE TABLE Driver(
  DRIVER_ID  BIGINT PRIMARY KEY AUTO_INCREMENT,
  Driver_Name VARCHAR(256),
  Route_Location VARCHAR(30), 
  ORDER_ID INT,
  FOREIGN KEY(ORDER_ID) REFERENCES STAFF(ORDER_ID)
);
